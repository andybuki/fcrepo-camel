46  /**
47   * Converts a Fedora Message into a format suitable for a LDN receiver.
48   * See: http://www.w3.org/TR/ldn/
49   *
50   * @author acoburn
51   */
52  public class LdnProcessor implements Processor {
53  
54      private static final String PROV = "http://www.w3.org/ns/prov#";
55  
56      private static final Property wasAssociatedWith = createProperty(PROV + "wasAssociatedWith");
57      private static final Property wasAttributedTo = createProperty(PROV + "wasAttributedTo");
58      private static final Property wasGeneratedBy = createProperty(PROV + "wasGeneratedBy");
59      private static final Property used = createProperty(PROV + "used");
60  
61      /**
62       * Process the Fedora message
63       *
64       * @param exchange the current camel message exchange
65       */
66      public void process(final Exchange exchange) throws IOException, NoSuchHeaderException {
67          final Message in = exchange.getIn();
68          final Model model = createDefaultModel();
69          final Model newModel = createDefaultModel();
70          final Resource resource = createResource(getSubjectUri(exchange));
71          final Resource event = createResource("");
72          final AtomicInteger counter = new AtomicInteger();
73          final ByteArrayOutputStream serializedGraph = new ByteArrayOutputStream();
74  
75          read(model, in.getBody(InputStream.class),
76                  contentTypeToLang(parse(in.getHeader(CONTENT_TYPE, String.class)).getMimeType()));
77  
78          newModel.add(createStatement(event, used, resource));
79          model.listObjectsOfProperty(resource, wasGeneratedBy).forEachRemaining(obj -> {
80              if (obj.isResource()) {
81                  obj.asResource().listProperties().forEachRemaining(stmt -> {
82                      newModel.add(createStatement(event, stmt.getPredicate(), stmt.getObject()));
83                  });
84              }
85          });
86          model.listObjectsOfProperty(resource, wasAttributedTo).forEachRemaining(obj -> {
87              final Resource agent = createResource("#agent" + Integer.toString(counter.getAndIncrement()));
88              if (obj.isResource()) {
89                  obj.asResource().listProperties().forEachRemaining(stmt -> {
90                      newModel.add(createStatement(agent, stmt.getPredicate(), stmt.getObject()));
91                  });
92              }
93              newModel.add(createStatement(event, wasAssociatedWith, agent));
94          });
95  
96          write(serializedGraph, newModel, JSONLD);
97          in.setBody(serializedGraph.toString("UTF-8"));
98          in.setHeader(HTTP_METHOD, "POST");
99          in.setHeader(CONTENT_TYPE, "application/ld+json");
100     }
101 }
