016package org.fcrepo.camel.processor;
017
018import java.io.IOException;
019
020import org.apache.camel.Exchange;
021import org.apache.camel.Message;
022import org.apache.camel.Processor;
023import org.fcrepo.camel.FcrepoHeaders;
024
025/**
026 * Represends a message processor that deletes objects from an
027 * external triplestore.
028 *
029 * @author Aaron Coburn
030 * @since Nov 8, 2014
031 */
032public class SparqlDeleteProcessor implements Processor {
033    /**
034     * Define how the message should be processed.
035     *
036     * @param exchange the current camel message exchange
037     */
038    public void process(final Exchange exchange) throws IOException {
039
040        final Message in = exchange.getIn();
041        final String namedGraph = in.getHeader(FcrepoHeaders.FCREPO_NAMED_GRAPH, String.class);
042        final String subject = ProcessorUtils.getSubjectUri(in);
043
044        /*
045         * N.B. The Sparql update command below deletes all triples with
046         * the defined subject uri (coming from the FCREPO_IDENTIFIER
047         * and FCREPO_BASE_URL headers). It also deletes triples that
048         * have a subject corresponding to that Fcrepo URI plus the
049         * "/fcr:export?format=jcr/xml" string appended to it. This
050         * makes it possible to more completely remove any triples
051         * for a given resource that were added earlier. If fcrepo
052         * ever stops producing triples that are appended with
053         * /fcr:export?format..., then that extra line can be removed.
054         * It would also be possible to recursively delete triples
055         * (by removing any triple whose subject is also an object of
056         * the starting (or context) URI, but that approach tends to delete
057         * too many triples from the triplestore. This command does
058         * not remove blank nodes.
059         */
060        final StringBuilder query = new StringBuilder("update=");
061
062        query.append(ProcessorUtils.deleteWhere(subject, namedGraph));
063        query.append(";\n");
064        query.append(ProcessorUtils.deleteWhere(subject + "/fcr:export?format=jcr/xml", namedGraph));
065
066        in.setBody(query.toString());
067        in.setHeader(Exchange.HTTP_METHOD, "POST");
068        in.setHeader(Exchange.CONTENT_TYPE, "application/x-www-form-urlencoded");
069   }
070
071}
